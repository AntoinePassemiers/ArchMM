from archmm.trees.tree import ClassificationTree
import numpy as np

# Generate random training samples
data = np.random.rand(2500, 25)
targets = np.random.randint(0, 10, size = 2500)

# Grow a tree according to these samples
tree = ClassificationTree(max_nodes = 1000)
tree.applyID3(data, targets)

# Generate new samples, emit predictions and update the tree
udata = np.random.rand(400, 25)
utargets = np.random.randint(0, 10, size = 400)
predictions = tree.classify(udata)
tree.update(udata, utargets)
